CREATE OR REPLACE TRIGGER T_BIU_INVOICES_TAB_CALCULATE_GROSS_VALUE
BEFORE INSERT OR UPDATE ON INVOICES_TAB
FOR EACH ROW
BEGIN
    IF INSERTING OR UPDATING THEN
        :NEW.INVO_GROSS_VALUE := ROUND(:NEW.INVO_NET_VALUE + (:NEW.INVO_NET_VALUE * :NEW.INVO_TAX_RATE_PCT / 100), 2);
    END IF;
END;
